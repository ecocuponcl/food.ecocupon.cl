**Modificar kiosk.js** — Agregar función scanQR():
```javascript
let html5QrcodeScanner;

async function initQRScanner() {
    const container = document.getElementById('qr-scanner-container');
    if (!container) return;
    
    html5QrcodeScanner = new Html5Qrcode("qr-reader");
    
    const config = {
        fps: 10,
        qrbox: { width: 250, height: 250 },
        aspectRatio: 1.0
    };
    
    html5QrcodeScanner.start(
        { facingMode: "environment" },
        config,
        onQRCodeSuccess,
        onQRCodeError
    ).catch(err => console.error("QR init error:", err));
}

async function onQRCodeSuccess(decodedText) {
    // Validar QR con backend
    const response = await fetch('/recycle/validate', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ qr_code: decodedText })
    });
    const result = await response.json();
    
    if (result.valid) {
        // Agregar al carrito como item reciclable
        addToCart({
            name: `${result.material} (${result.weight}kg)`,
            price: -result.cashback, // Cashback negativo = descuento
            cashback: result.cashback,
            material: result.material,
            weight: result.weight,
            isRecycle: true
        });
        
        // Feedback visual
        showNotification(`✅ ${result.material} — $${result.cashback.toLocaleString('es-CL')} CLP acreditado`);
        
        // Pausar scanner
        if (html5QrcodeScanner) html5QrcodeScanner.pause();
    } else {
        showNotification(`❌ QR inválido: ${result.error}`);
    }
}
```

### TAREA 3: Mostrar QR después del pago

**Archivo**: `food_kiosk/controllers/kiosk_controller.py`

Después de confirmar pago Flow, generar y mostrar QR codes:

```python
@http.route(['/kiosk/payment_webhook'], type='json', auth='public', methods=['POST'], csrf=False)
def payment_webhook(self, **kwargs):
    # ... existing payment logic ...
    
    # Generar QR codes para items reciclables
    order_items = request.session.get('cart_items', [])
    qr_codes = []
    
    for item in order_items:
        if item.get('is_recyclable'):
            # Llamar a /decide del agent
            response = requests.post('http://localhost:9000/decide', json={
                'query': f"Reciclar {item['name']} {item['weight']}kg",
                'vertical': 'RECYCLE',
                'metadata': {'material': item['material'], 'weight': item['weight']}
            })
            result = response.json()
            if result.get('qr_code'):
                qr_codes.append(result)
    
    # Guardar en sesión para mostrar en página de retorno
    request.session['qr_codes'] = qr_codes
    return {'status': 'ok', 'qr_codes': qr_codes}
```
